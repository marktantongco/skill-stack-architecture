   282→function CodeBlock({ code, label }: { code: string; label?: string }) {
   283→  const [copied, setCopied] = useState(false);
   284→  const handleCopy = async () => {
   285→    const ok = await copyToClipboard(code);
   286→    if (ok) { setCopied(true); setTimeout(() => setCopied(false), 2000); }
   287→  };
   288→  return (
   289→    <div className="relative group">
   290→      {label && <span className="text-[9px] tracking-[0.2em] uppercase text-muted-foreground font-mono">{label}</span>}
   291→      <pre className="bg-muted/60 border border-border rounded p-3 text-xs font-mono overflow-x-auto custom-scrollbar text-foreground/80 mt-1">
   292→        <code>{code}</code>
   293→      </pre>
   294→      <button onClick={handleCopy} className="absolute top-2 right-2 min-w-[32px] min-h-[32px] flex items-center justify-center rounded hover:bg-muted cursor-pointer" aria-label="Copy code">
   295→        {copied ? <Check className="w-3.5 h-3.5 text-emerald-500" /> : <Copy className="w-3.5 h-3.5 text-muted-foreground" />}
   296→      </button>
   297→    </div>
   298→  );
   299→}

   979→// SUBPAGE: ARCHITECTURE
   980→// ═══════════════════════════════════════════════════════════
   981→function ArchitectureSubpage() {
   982→  const shouldReduce = useReducedMotion();
   983→  const [animCount, setAnimCount] = useState(0);
   984→
   985→  // Animate counter on mount
   986→  useEffect(() => {
   987→    let frame = 0;
   988→    const target = telemetryMetrics.totalInvocations;
   989→    const duration = 1500;
   990→    const start = performance.now();
   991→    const step = (now: number) => {
   992→      const elapsed = now - start;
   993→      const progress = Math.min(elapsed / duration, 1);
   994→      const eased = 1 - Math.pow(1 - progress, 3);
   995→      setAnimCount(Math.round(eased * target));
   996→      if (progress < 1) frame = requestAnimationFrame(step);
   997→    };
   998→    frame = requestAnimationFrame(step);
   999→    return () => cancelAnimationFrame(frame);
  1000→  }, []);
  1001→
  1002→  return (
