   192→        <code>{code}</code>
   193→      </pre>
   194→      <button onClick={handleCopy} className="absolute top-2 right-2 min-w-[32px] min-h-[32px] flex items-center justify-center rounded hover:bg-muted cursor-pointer" aria-label="Copy code">
